# 妙投神透 H5 实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 构建一个彩票号码生成 H5 应用，包含双色球和大乐透两种玩法，4 个页面（2 主页 + 2 结果页），纯静态部署。

**架构：** Vue 3 + Vite + Tailwind CSS 单页应用，通过 Vue Router 管理路由，号码通过 sessionStorage 传递，使用 html2canvas 实现图片保存。

**技术栈：** Vue 3 (Composition API) + TypeScript + Vite + Tailwind CSS + Vue Router + html2canvas

```typescript
export interface SSQResult {
  red: number[]
  blue: number[]
}

export interface DLTResult {
  front: number[]
  back: number[]
}

export interface LotteryMeta {
  type: 'ssq' | 'dlt'
  notes: number
  mode: 'single' | 'multiple' | 'dantuo'
  time: string
}

/**
 * 八卦时辰随机方案
 * Math.random() + 时辰偏移
 */
export function randomWithBagua(min: number, max: number, count: number): number[] {
  const range = max - min + 1
  const hour = new Date().getHours()
  const baguaOffset = Math.floor(hour / 3) % 8

  const result = new Set<number>()
  while (result.size < count) {
    const rand = Math.floor(Math.random() * range) + min
    const shifted = ((rand + baguaOffset - 1) % range) + min
    result.add(shifted)
  }
  return Array.from(result).sort((a, b) => a - b)
}

/**
 * 生成双色球号码
 */
export function generateSSQ(notes: number): SSQResult[] {
  return Array.from({ length: notes }, () => ({
    red: randomWithBagua(1, 33, 6),
    blue: randomWithBagua(1, 16, 1),
  }))
}

/**
 * 生成大乐透号码
 */
export function generateDLT(notes: number): DLTResult[] {
  return Array.from({ length: notes }, () => ({
    front: randomWithBagua(1, 35, 5),
    back: randomWithBagua(1, 12, 2),
  }))
}

/**
 * 格式化当前时间
 */
export function formatTime(): string {
  const now = new Date()
  const pad = (n: number) => String(n).padStart(2, '0')
  return `${now.getFullYear()}-${pad(now.getMonth() + 1)}-${pad(now.getDate())} ${pad(now.getHours())}:${pad(now.getMinutes())}:${pad(now.getSeconds())}`
}

/**
 * 获取当前期号（模拟）
 */
export function getIssueNumber(type: 'ssq' | 'dlt'): string {
  const now = new Date()
  const year = now.getFullYear()
  const shortYear = String(year).slice(-2)
  // 模拟期号：年份后2位 + 当年第几周
  const start = new Date(year, 0, 1)
  const week = Math.ceil(((now.getTime() - start.getTime()) / 86400000 + start.getDay() + 1) / 7)
  const prefix = type === 'ssq' ? '' : ''
  return `第${year}${String(week).padStart(2, '0')}期`
}
```

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useRouter, useRoute } from 'vue-router'
import html2canvas from 'html2canvas'
import ResultCard from '@/components/ResultCard.vue'
import { generateSSQ, generateDLT, formatTime, getIssueNumber } from '@/composables/useLottery'

const router = useRouter()
const route = useRoute()

const lotteryType = ref<'ssq' | 'dlt'>((route.query.type as 'ssq' | 'dlt') || 'ssq')
const notes = Number(route.query.notes) || 1
const mode = (route.query.mode as string) || 'single'

const resultCardRef = ref<HTMLElement | null>(null)
const numbers = ref<any[]>([])
const issueNumber = ref('')
const metaInfo = ref('')

const blessings = {
  ssq: {
    main: '🎉 恭喜发财 · 好运连连 🎉',
    sub: '五行八卦 · 吉时已到 · 必中大奖',
  },
  dlt: {
    main: '🎊 吉星高照 · 鸿运当头 🎊',
    sub: '奇门遁甲 · 神机妙算 · 必得头奖',
  },
}

const modeLabels = {
  single: '单式',
  multiple: '复式',
  dantuo: '胆拖',
}

onMounted(() => {
  // 生成号码
  if (lotteryType.value === 'ssq') {
    numbers.value = generateSSQ(notes)
  } else {
    numbers.value = generateDLT(notes)
  }

  issueNumber.value = getIssueNumber(lotteryType.value)
  metaInfo.value = `生成时间：${formatTime()}  注数：${notes}注 · 模式：${modeLabels[mode as keyof typeof modeLabels]}`
})

const primaryColor = lotteryType.value === 'ssq' ? '#EF4444' : '#3B82F6'

async function handleSaveImage() {
  if (!resultCardRef.value) return

  try {
    const canvas = await html2canvas(resultCardRef.value, {
      backgroundColor: '#FFF8F0',
      scale: 2,
    })
    const link = document.createElement('a')
    link.download = `妙投神透-${lotteryType.value === 'ssq' ? '双色球' : '大乐透'}-${Date.now()}.png`
    link.href = canvas.toDataURL('image/png')
    link.click()
  } catch (e) {
    alert('保存图片失败，请长按截图')
  }
}

function handleBack() {
  router.push({ path: '/', query: { type: lotteryType.value } })
}
</script>

<template>

    <!-- 顶部导航栏 -->
    <header class="bg-white px-6 py-4">
      <div class="flex items-center gap-2">
        <button class="text-[#6B7280]" @click="handleBack">
          <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
            <path d="M19 12H5M12 19l-7-7 7-7" />
          </svg>
        </button>
        <h1 class="text-lg font-bold text-[#1F2937]">
          {{ lotteryType === 'ssq' ? '双色球结果' : '大乐透结果' }}
        </h1>
      </div>
    </header>

    <!-- 号码展示区 -->
    <main class="flex-1 px-6 py-16" ref="resultCardRef">
      <ResultCard
        :lottery-type="lotteryType"
        :numbers="numbers"
        :issue-number="issueNumber"
        :meta="metaInfo"
      />

      <!-- 祝福语 -->
      <div class="mt-8 text-center space-y-2">
        <p class="text-lg font-semibold" :class="lotteryType === 'ssq' ? 'text-[#D97706]' : 'text-[#1D4ED8]'">
          {{ blessings[lotteryType].main }}
        </p>
        <p class="text-base" :class="lotteryType === 'ssq' ? 'text-[#92400E]' : 'text-[#1E40AF]'">
          {{ blessings[lotteryType].sub }}
        </p>
      </div>
    </main>

    <!-- 底部操作栏 -->
    <footer class="px-6 pb-10">
      <div class="grid grid-cols-2 gap-4">
        <button
          class="h-12 rounded-xl text-white font-medium shadow-lg flex items-center justify-center gap-2"
          :class="lotteryType === 'ssq' ? 'bg-[#EF4444]' : 'bg-[#3B82F6]'"

        >
          <svg xmlns="http://www.w3.org/2000/svg" width="22" height="22" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
            <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4" /><polyline points="7 10 12 15 17 10" /><line x1="12" y1="15" x2="12" y2="3" />
          </svg>
          保存图片
        </button>
        <button
          class="h-12 rounded-xl font-medium shadow-lg border-2 flex items-center justify-center gap-2"
